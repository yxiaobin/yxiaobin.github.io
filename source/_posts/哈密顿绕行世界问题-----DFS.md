思路：  
比赛的时候是不傻了！！！！！！只是记录了进行了20次dfs的数据，怎么就没有想到去把第20次遍历的不是m的遍历给去掉！！！  

    
    
    #include <iostream>
    #include <cstdio>
    #include <cstring>
    #include <cmath>
    #include <vector>
    #include <queue>
    using namespace std;
    vector<int>v[150];
    int vis[100];
    int sas[100];
    int tt,m;
    void dfs(int x, int num)
    {
        sas[num]=x;
        if(num==20&& x==m)
        {
            printf("%d: ",tt++);
            for(int i=0; i<=num; i++)
            {
                printf(" %d",sas[i]);
            }
            printf("\n");
            return ;
        }
        else
        {
            for(int i=0; i<v[x].size(); i++)
            {
                int id=v[x][i];
                if(!vis[id])
                {
                    vis[id]=1;
                    dfs(id,num+1);
                    vis[id]=0;
                }
            }
        }
    }
    int main ()
    {
        int a,b,c;
        for(int i=1; i<=20; i++)
        {
            cin>>a>>b>>c;
            v[i].push_back(a);
            v[i].push_back(b);
            v[i].push_back(c);
        }
        tt=1;
        while(cin>>m && m)
        {
            memset(vis,0,sizeof(vis));
            dfs(m,0);
        }
    }
