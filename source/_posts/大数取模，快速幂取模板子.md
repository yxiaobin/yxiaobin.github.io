整数快速幂取模：

    
    
        long mi(long a,long k,long M)  
        {  
            long b=1;  
            while(k>=1){  
                if(k%2==1){  
                    b=a*b%M;  
                }  
                a=(a%M)*(a%M)%M;  
                k/=2;  
            }  
            return b;  
        }
